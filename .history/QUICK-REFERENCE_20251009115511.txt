  ┌─────────────────────────────────────────────────────────────────────┐
  │                                                                       │
  │  import { useOptimizedCategories }                                   │
  │    from '@/features/lats/hooks/useOptimizedCategories';              │
  │                                                                       │
  │  function MyComponent() {                                            │
  │    const { categories, loading } = useOptimizedCategories();         │
  │                                                                       │
  │    if (loading) return <div>Loading...</div>;                        │
  │                                                                       │
  │    return (                                                          │
  │      <div>                                                           │
  │        {categories.map(cat => (                                      │
  │          <div key={cat.id}>                                          │
  │            {cat.icon} {cat.name}                                     │
  │          </div>                                                      │
  │        ))}                                                           │
  │      </div>                                                          │
  │    );                                                                │
  │  }                                                                   │
